		 * Descrition:
		 * This project is a traffic light simulation that switches the state
		 * of the lights in green per 5 seconds, yellow per 3 seconds and red per
		 * 8 seconds.
		 */

		//Declaration of variables
		#define red_light 2     //defines that pin two is the red light
		#define yellow_light 3  //defines that pin three is the yellow light 
		#define green_light 4   //defines that pin four is the green light

		void setup() {
		  for (byte light=2; light<5; light++){
		    pinMode(light, OUTPUT);     //Set Arduino ports as OUTPUT
		    //Start of lights tests
		    digitalWrite(light, HIGH);
		    delay(1000);
		    digitalWrite(light, LOW);
		    //End of lights tests
		  }
		}

		void loop() {
		  trafficLight(1, 0, 0, 5000); //Green light per 5 seconds
		  trafficLight(0, 1, 0, 3000); //Yellow light per 3 seconds
		  trafficLight(0, 0, 1, 8000); //Red light per 8 seconds
		}

		void trafficLight(bool green, bool yellow, bool red, int time_light){
		  digitalWrite(red_light, red);       //Turn On or Off the red light
		  digitalWrite(yellow_light, yellow); //Turn On or Off the yellow light
		  digitalWrite(green_light, green);   //Turn On or Off the green light
		  delay(time_light);                  //Wait time of the light traffic
		}
